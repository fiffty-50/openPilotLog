66  for (int i = 0; i < 7; i++) { //\033[32m
67  if (validationArray[i])
68  deb_string += "\t\033[32m" + QString::number(validationArray[i]);

70  deb_string += "\t\033[31m" + QString::number(validationArray[i]);
